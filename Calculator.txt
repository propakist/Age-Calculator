    <script>
        // Set max date to today to prevent future dates in picker
        const dateInput = document.getElementById('birthdate');
        dateInput.max = new Date().toISOString().split("T")[0];

        function calculateAge() {
            const birthInput = document.getElementById('birthdate').value;
            const errorMsg = document.getElementById('error-msg');
            const resultBox = document.getElementById('result');

            // Validation
            if (!birthInput) {
                errorMsg.style.display = 'block';
                errorMsg.innerText = "Please enter your birth date.";
                resultBox.style.display = 'none';
                return;
            }

            const birthDate = new Date(birthInput);
            const today = new Date();

            // Check if future date (double check)
            if (birthDate > today) {
                errorMsg.style.display = 'block';
                errorMsg.innerText = "Date cannot be in the future.";
                resultBox.style.display = 'none';
                return;
            }

            errorMsg.style.display = 'none';

            let y1 = birthDate.getFullYear();
            let m1 = birthDate.getMonth() + 1;
            let d1 = birthDate.getDate();

            let y2 = today.getFullYear();
            let m2 = today.getMonth() + 1;
            let d2 = today.getDate();

            let d3, m3, y3;

            y3 = y2 - y1;

            if (m2 >= m1) {
                m3 = m2 - m1;
            } else {
                y3--;
                m3 = 12 + m2 - m1;
            }

            if (d2 >= d1) {
                d3 = d2 - d1;
            } else {
                m3--;
                if (m3 < 0) {
                    m3 = 11;
                    y3--;
                }
                d3 = getDaysInMonth(y1, m1) + d2 - d1;
            }

            // Update UI
            document.getElementById('years').innerText = y3;
            document.getElementById('months').innerText = m3;
            document.getElementById('days').innerText = d3;
            
            resultBox.style.display = 'block';
        }

        function getDaysInMonth(year, month) {
            return new Date(year, month, 0).getDate();
        }
    </script>
